	\textit{Statistical models linking explanatory variables to the presence of densification}
	
	\medskip
	
	\textbf{Contributions}
	
	$\rightarrow$ modelling with heterogeneous data across 3 countries
	
	$\rightarrow$ spatial statistics for densification
	
	$\rightarrow$ sensitivity analysis
	
	\medskip
	
	\textbf{Takeaways}
	
	$\rightarrow$ varying explanatory power across countries
	
	$\rightarrow$ strong effect of spatial lag and spatial structures like core-periphery
	
	$\rightarrow$ no generic rules, need for tailored planning policies

	\textit{A novel algorithm for matching combining GMoA and MCA into a multi-modelling approach}
	
	\medskip
	
	\textbf{Contributions}
	
	$\rightarrow$ novel algorithm performing better than state-of-the-art for change detection
	
	$\rightarrow$ validated and benchamrked on synthetic data and ground-truth datasets
	
	$\rightarrow$ an open python library for matching \url{https://github.com/umrlastig/pymatch}

	\textbf{Takeways}
